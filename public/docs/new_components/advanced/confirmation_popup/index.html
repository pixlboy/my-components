Confirmation Pop Up demo
Explanation of component
  1. confirmation popup component which has title,body content, close and can have n no.of buttons like cancel, add, delete, submit
  2. click button and modal box opens with title, content, buttons
  3. close button to the top right to close the whole tabbed wrapper component based on visible true or false (because we need to open it again)
  4. modalbox positioned center of the page
Usage of Component
  1. step1: Create config with title, buttons, and content: 
    						var config = {
							title: "",
							content: "",
							buttons: ['cancel', 'add', 'submit']
						}
  2. step2: Render modalbox - popup confirmation 
    						var onClose = function(visible){
							JnprCL.ConfirmationPopUpStore('appId_1').setVisible( true );
						}

						var onBtnClicked = function(visible){
							JnprCL.ConfirmationPopUpStore('appId_1').setVisible( true );
						}

						var modalPopUp = function(config){
							JnprCL.ConfirmationPopUpStore('appId_1').setConfig(
								{
									title: "Title Anything can go here",
									content: "any content can go here asdasdasdas dsadad adasdad asdasdasd asdasdas sdasdasd yewhwehdxmncmsacn  ajcsnakmdasd jsadajsdnas ajsdnakjsdnaksj ",
									buttons: [
										{
											buttonId:'buttonid',
											title:'cancel',
											status:'status'
										},
										{
											buttonId:'buttonid',
											title:'Add',
											status:'Active'
										}
									],
									visible:  true
							})


						}

						ReactDOM.render(React.createElement(ConfirmationPopUpComp, {
							appId: "appId_1",
						    onClose: this.onClose.bind(this),
							config: config,

						}), document.getElementById('popup'));
